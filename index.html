# ===================== KIKO'S BURGUER - COST MANAGER ===================== from flask import Flask, request, redirect, render_template_string import sqlite3 from decimal import Decimal app = Flask(__name__) # ===================== DATABASE SETUP ===================== def get_db(): conn = sqlite3.connect("burger_kiko.db") conn.row_factory = sqlite3.Row return conn def init_db(): conn = get_db() cur = conn.cursor() cur.execute("CREATE TABLE IF NOT EXISTS ingredientes (id INTEGER PRIMARY KEY AUTOINCREMENT, nombre TEXT, precio REAL, unidad TEXT)") cur.execute("CREATE TABLE IF NOT EXISTS productos (id INTEGER PRIMARY KEY AUTOINCREMENT, nombre TEXT, packaging REAL, delivery REAL)") cur.execute("CREATE TABLE IF NOT EXISTS producto_ingrediente (producto_id INTEGER, ingrediente_id INTEGER, cantidad REAL)") conn.commit() conn.close() init_db() # ===================== CALCULATIONS ===================== def obtener_costos_todos(): conn = get_db() cur = conn.cursor() cur.execute("SELECT * FROM productos") prods = cur.fetchall() resultados = [] for p in prods: cur.execute(""" SELECT pi.cantidad, i.precio FROM producto_ingrediente pi JOIN ingredientes i ON pi.ingrediente_id = i.id WHERE pi.producto_id=?""", (p['id'],)) costo_ingredientes = sum(Decimal(str(r[0])) * Decimal(str(r[1])) for r in cur.fetchall()) total = costo_ingredientes + Decimal(str(p['packaging'])) + Decimal(str(p['delivery'])) resultados.append({ 'id': p['id'], 'nombre': p['nombre'], 'costo_base': float(costo_ingredientes), 'total': float(total) }) conn.close() return resultados # ===================== MODERN UI TEMPLATE ===================== TEMPLATE = """ Kiko's Burguer Admin
🍔 KIKO'S BURGUER - CONTROL DE COSTOS
Nuevo Ingrediente
$
Despensa de Precios
    {% for i in ingredientes %}
  • {{ i.nombre }} ${{ i.precio }} / {{ i.unidad }}
    • {% endfor %}
Crear Nueva Hamburguesa
Ingredientes (ID=Cantidad, ej: 1=150):
Listado de Costos de Venta
{% for p in costos %} {% endfor %}
Producto Costo Ingredientes Costo Final (c/Pack) Acción
{{ p.nombre }} ${{ "%.2f"|format(p.costo_base) }} ${{ "%.2f"|format(p.total) }}
""" # ===================== ROUTES ===================== @app.route('/') def index(): conn = get_db() ingredientes = conn.execute("SELECT * FROM ingredientes").fetchall() costos = obtener_costos_todos() conn.close() return render_template_string(TEMPLATE, ingredientes=ingredientes, costos=costos) @app.route('/add_ing', methods=['POST']) def add_ing(): conn = get_db() conn.execute("INSERT INTO ingredientes (nombre, precio, unidad) VALUES (?, ?, ?)", (request.form['nombre'], float(request.form['precio']), request.form['unidad'])) conn.commit() conn.close() return redirect('/') @app.route('/add_prod', methods=['POST']) def add_prod(): conn = get_db() cur = conn.cursor() cur.execute("INSERT INTO productos (nombre, packaging, delivery) VALUES (?, ?, ?)", (request.form['nombre'], float(request.form['packaging'] or 0), float(request.form['delivery'] or 0))) pid = cur.lastrowid for item in request.form['ingredientes'].split(','): if '=' in item: i, c = item.split('=') cur.execute("INSERT INTO producto_ingrediente VALUES (?, ?, ?)", (pid, int(i), float(c))) conn.commit() conn.close() return redirect('/') if __name__ == '__main__': app.run(debug=True)
